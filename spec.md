`fishes.c`:

```c
#include <stdio.h>
#include <time.h>
#include <stdlib.h>
#define XQ 500
#define YQ 500
#define FB 10   // Порог размножения рыб
#define SB 50   // Порог размножения акул
#define SH 15   // Лимит голодания акул
#define GQ 3000 // Максимум поколений

typedef struct {
    char nm;    // 'w' - вода, 'f' - рыба, 's' - акула
    int bc;     // Счетчик возраста (размножение)
    int hc;     // Счетчик голода (только для акул)
    int flag;   // Номер текущего поколения (защита от двойного хода)
} cell;

cell sea[YQ][XQ];
int fqnt = 0;
int sqnt = 0;

void step(int y, int x, int g) {
    // Если ячейка уже ходила в этом поколении или это вода — выходим
    if (sea[y][x].flag == g || sea[y][x].nm == 'w') {
        return;
    }

    int tr[4][2]; 
    int tq = 0;
    int dy[] = {-1, 1, 0, 0};
    int dx[] = {0, 0, -1, 1};

    // Фиксируем ход и инкрементируем возраст
    sea[y][x].flag = g;
    sea[y][x].bc++;

    char primary_target = (sea[y][x].nm == 's') ? 'f' : 'w';
    
    // Поиск основной цели (рыба для акулы, вода для рыбы)
    for (int i = 0; i < 4; i++) {
        int ny = (y + dy[i] + YQ) % YQ;
        int nx = (x + dx[i] + XQ) % XQ;
        
        if (sea[ny][nx].nm == primary_target) {
            tr[tq][0] = ny;
            tr[tq][1] = nx;
            tq++;
        }
    }
    
    // Если акула не нашла еду, она ищет воду для перемещения
    if (sea[y][x].nm == 's' && tq == 0) {
        sea[y][x].hc++; // Увеличиваем голод
        for (int i = 0; i < 4; i++) {
            int ny = (y + dy[i] + YQ) % YQ;
            int nx = (x + dx[i] + XQ) % XQ;
            
            if (sea[ny][nx].nm == 'w') {
                tr[tq][0] = ny;
                tr[tq][1] = nx;
                tq++;
            }
        }
    } else if (sea[y][x].nm == 's') {
        // Если акула нашла еду, голод сбрасывается
        sea[y][x].hc = 0;
    }

    // Смерть акулы от голода
    if (sea[y][x].nm == 's' && sea[y][x].hc >= SH) {
        sea[y][x] = (cell){'w', 0, 0, 0};
        sqnt--;
        return;
    }

    // Логика перемещения и размножения
    if (tq > 0) {
        int r = rand() % tq;
        int ny = tr[r][0];
        int nx = tr[r][1];

        // Если на целевой клетке была рыба — она съедена
        if (sea[ny][nx].nm == 'f') {
            fqnt--;
        }

        char type = sea[y][x].nm;
        int reproduction_limit = (type == 'f') ? FB : SB;

        // Перемещаем особь в новую ячейку
        sea[ny][nx] = sea[y][x];

        if (sea[ny][nx].bc >= reproduction_limit) {
            // Размножение: обнуляем возраст и оставляем потомка в старой ячейке
            sea[ny][nx].bc = 0;
            sea[y][x] = (cell){type, 0, 0, g};
            
            if (type == 'f') {
                fqnt++;
            } else {
                sqnt++;
            }
        } else {
            // Обычное перемещение: старая ячейка становится водой
            sea[y][x] = (cell){'w', 0, 0, 0};
        }
    }
}

int main() {
    srand(time(NULL));
    FILE *f = fopen("results.csv", "w");
    if (f == NULL) {
        return 1;
    }
    
    fprintf(f, "t,fish,sharks\n");

    // Инициализация поля с рандомным распределением и случайным возрастом
    for (int y = 0; y < YQ; y++) {
        for (int x = 0; x < XQ; x++) {
            double r = rand() / (double)RAND_MAX;
            
            if (r < 0.10) { 
                // Создаем рыбу со случайным начальным возрастом (от 0 до FB-1)
                sea[y][x] = (cell){'f', rand() % FB, 0, 0}; 
                fqnt++; 
            } else if (r < 0.12) { 
                // Создаем акулу со случайным возрастом и голодом
                sea[y][x] = (cell){'s', rand() % SB, rand() % SH, 0}; 
                sqnt++; 
            } else {
                sea[y][x] = (cell){'w', 0, 0, 0};
            }
        }
    }

    // Основной цикл симуляции
    for (int g = 0; g <= GQ && (fqnt + sqnt) > 0; g++) {
        // Записываем текущее состояние ПЕРЕД ходом (начиная с поколения 0)
        fprintf(f, "%d,%d,%d\n", g, fqnt, sqnt);

        // Обход поля
        for (int y = 0; y < YQ; y++) {
            for (int x = 0; x < XQ; x++) {
                step(y, x, g + 1); // g+1 чтобы флаг ячейки был всегда актуальным
            }
        }
    }

    fclose(f);
    return 0;
}
```

```py
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# 1. Загрузка данных
# Укажи путь к своему CSV файлу
df = pd.read_csv('results.csv')

# 2. Настройки анализа
BURN_IN = 500  # Отбрасываем первые 500 поколений (переходный процесс)
WINDOW = 50   # Окно сглаживания (подбирается экспериментально)

# Отрезаем начальный хаос
df_steady = df[df['t'] > BURN_IN].copy()

# 3. Автоматический расчет эмпирических стационарных точек
mean_fish = df_steady['fish'].mean()
mean_sharks = df_steady['sharks'].mean()

print("=== АВТОМАТИЧЕСКИЙ АНАЛИЗ ===")
print(f"Эмпирическая точка равновесия (в среднем особей):")
print(f"Рыбы (X_avg): {mean_fish:.2f}")
print(f"Акулы (Y_avg): {mean_sharks:.2f}")

# 4. Сглаживание данных для графиков
df_steady['fish_smooth'] = df_steady['fish'].rolling(window=WINDOW).mean()
df_steady['sharks_smooth'] = df_steady['sharks'].rolling(window=WINDOW).mean()

# Очищаем NaN, которые появились из-за скользящего окна
df_steady = df_steady.dropna()

# 5. Построение сглаженного фазового портрета
plt.figure(figsize=(8, 6))
plt.plot(df_steady['fish_smooth'],
         df_steady['sharks_smooth'], color='blue', linewidth=1)
plt.scatter([mean_fish], [mean_sharks], color='red',
            marker='X', s=100, label='Эмпирическое равновесие')

plt.title('Сглаженный фазовый портрет популяций')
plt.xlabel('Численность рыб (сглаженная)')
plt.ylabel('Численность акул (сглаженная)')
plt.legend()
plt.grid(True)
plt.show()
```

```md
# Детальный анализ модификаций и оптимизаций исходного кода (Модель Ва-Тор)

В данном документе подробно описаны все изменения, внесенные в базовую реализацию клеточного автомата Ва-Тор (исходный файл `fishes.c`). Цель модификаций — приведение программы в соответствие с математическими требованиями лабораторной работы, повышение производительности и устранение логических ошибок исходной архитектуры.

## 1. Архитектурные изменения и сбор данных

* **Удаление графической подсистемы (SDL2):** Исходный код был жестко привязан к библиотеке SDL2 для покадровой отрисовки состояний в графическом окне. Весь этот функционал (включая зависимости `sdl_fb.h`, цветовые константы, задержки `sleep_ms`) был полностью вырезан. Это перевело программу из разряда интерактивных графических приложений в разряд высокоскоростных вычислительных симуляций.

* **Внедрение файлового логгирования:** В соответствии с требованиями методических указаний, добавлен вывод данных в формат CSV (`results.csv`). На каждом поколении фиксируется текущее время (шаг), численность рыб и численность акул. Это обеспечивает сырую базу временных рядов, необходимую для построения фазовых портретов и анализа особых точек (Седло, Центр).

## 2. Исправление логических уязвимостей

* **Корректировка вероятностного распределения:** В оригинальном коде логика появления особей при старте содержала грубую ошибку перекрытия интервалов:
  ```

  Вектор ветвления заменен на строгую последовательность непересекающихся порогов (`r < 0.10` для рыб, и `r < 0.12` для акул), что гарантирует корректную стартовую популяцию хищников.

## 3. Десинхронизация жизненных циклов (Сглаживание графиков)

* **Проблема «Ступенчатого» размножения:** В изначальной или наивной реализации все особи появляются с возрастом `bc = 0`. В результате, на шаге `FB` (для рыб) или `SB` (для акул) вся популяция синхронно дает потомство, удваивая численность за один такт. Это создает резкие, неестественные скачки на графиках численности.

* **Решение:** Введена жесткая рандомизация начальных параметров. При старте популяции возраст каждой рыбы устанавливается как `rand() % FB`, акулы — `rand() % SB`, а начальный голод акулы — `rand() % SH`. Это «размазывает» процесс размножения и смертности по времени, приводя графики к плавным колебаниям, характерным для классических дифференциальных уравнений Лотки-Вольтерры.

## 4. Оптимизация производительности (Снижение накладных расходов)

* Изолированная функция `nbr`, вычислявшая доступных соседей, упразднена.

* **Минимизация структуры данных:** Структура `cell` была очищена от неиспользуемых полей. Оставлены только минимально необходимые: тип (`nm`), возраст (`bc`), голод (`hc`) и метка поколения (`flag`).

## 5. Алгоритмическая корректность (Защита от «Бегущей волны»)

* **Однопроходное обновление:** В клеточных автоматах последовательный обход матрицы сверху-вниз может привести к тому, что переместившаяся вниз или вправо особь будет обработана дважды (или более раз) за один такт.

* **Реализация защиты:** Вместо выделения памяти под вторую матрицу (Double Buffering), используется поле `flag`. При обработке клетки `sea[y][x].flag` устанавливается равным номеру текущего поколения `g`. Если алгоритм натыкается на клетку, где `flag == g`, он её пропускает, так как она уже совершила действие на этом шаге. Это позволяет обновлять поле «на месте» (in-place), экономя 50% оперативной памяти и избегая лишних аллокаций.
